#Function to print confusion matrix

def confus_mat(y_test,pred):
    class_names = ['not_fraud', 'fraud']
    matrix = confusion_matrix(y_test, pred)
    # Create pandas dataframe
    dataframe = pd.DataFrame(matrix, index=class_names, columns=class_names)
    # Create heatmap
    sns.heatmap(dataframe, annot=True, cbar=None, cmap="Blues", fmt = 'g')
    plt.title("Confusion Matrix"), plt.tight_layout()
    plt.ylabel("True Class"), plt.xlabel("Predicted Class")
    plt.show()
    

#  For a financial institution dealing with identifying fraud, Sensitivity and F1 - Score might be more important metrics. 
# F1- Score reprsents a more balanced result as it is the harmonic mean between Precision and Recall. 
# Sensitivity is more important in the sense that we are more interested in identifying fraud than than identifying legitimate customers.
    
def f1(y_test,pred):
    f1_s = round(f1_score(y_test, pred), 2)
    r_score = round(recall_score(y_test, pred), 2)
    print("Sensitivity/Recall : {recall_score}".format(recall_score = r_score))
    print("F1 Score : {f1_score}".format(f1_score = f1_s))

# Creating a function, who accepts the Training and test data and perform various classifier on it

def classifier(X_train, X_test, y_train, y_test):
       
    # Logistic Regression
    lr = LogisticRegression()
    lr.fit(X_train, y_train)
    pred = lr.predict(X_test)
    confus_mat(y_test,pred)
    f1(y_test,pred)
    
    # Decison Tree Classifier
    clf = DecisionTreeClassifier().fit(X_train, y_train)
    pred = clf.predict(X_test)
    confus_mat(y_test,pred)
    f1(y_test,pred)
    
    # Random forest Classifier
    rfc = RandomForestClassifier(max_depth=2, random_state=0)
    rfc.fit(X_train, y_train)
    pred = rfc.predict(X_test)
    confus_mat(y_test,pred)
    f1(y_test,pred)
